<script>
const ENDPOINT = "YOUR_GOOGLE_APPS_SCRIPT_URL"; 
// thay bằng URL Apps Script của bạn

document.getElementById('submitBtn').addEventListener('click', async () => {
  const x = document.getElementById('xHandle').value.trim();
  const wallet = document.getElementById('wallet').value.trim();
  const msg = document.getElementById('msg');

  if (!x || !wallet) {
    msg.textContent = "Điền đủ X + Wallet nhé.";
    return;
  }

  msg.textContent = "Đang gửi...";

  try {
    const res = await fetch(ENDPOINT, {
      method: "POST",
      body: JSON.stringify({ x, wallet }),
      headers: { "Content-Type": "application/json" }
    });

    const j = await res.json();
    msg.textContent = (j.status === "ok")
      ? "Thành công — Welcome to UDI!"
      : "Lỗi: " + j.message;

  } catch (err) {
    msg.textContent = "Lỗi mạng: " + err;
  }
});
</script>
